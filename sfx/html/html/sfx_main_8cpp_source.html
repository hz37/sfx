244 void sfxFrame::OnSearchButtonClick(wxCommandEvent& event)
245 {

247 
248  wxString s;
249  s.Printf(_("%d results"), resultCount);

251 
252  // The result set can be quite lengthy, so freeze updates until we are done.
253 
254  ResultsListCtrl->Freeze();
255 
256  ResultsListCtrl->ClearAll();
257 
258  wxListItem col0;
259  col0.SetId(0);
260 
261  ResultsListCtrl->InsertColumn(0, col0);
262 
263  for(int idx = 0; idx < resultCount; ++idx)
264  {
265  wxListItem item;
266  item.SetId(idx);
267  ResultsListCtrl->InsertItem(item);
268  wxString description = m_model.searchedDescriptionAt(idx);
269  ResultsListCtrl->SetItem(idx, 0, description);
270  }
271 
272  ResultsListCtrl->SetColumnWidth(0, wxLIST_AUTOSIZE);
273 
274  // Unfreeze updates.
275 
276  ResultsListCtrl->Thaw();
277 }
