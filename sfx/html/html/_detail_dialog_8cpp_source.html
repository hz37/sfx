178 bool DetailDialog::CreateWavFile(wxString& fileName)
179 {
180  static int tempWavCounter = 0;
182  if(!m_waveData.get() || (m_selStart >= m_selStop))
183  {
186  return false;
187  }
188 
191  fileName << m_storageDirectory
192  << c_tempPrefix
193  << m_filePrefix
194  << tempWavCounter++
195  << ".wav";
196 
197  SF_INFO sfInfo;
198 
199  sfInfo.channels = m_channels;
200  sfInfo.format = m_format;
201  sfInfo.samplerate = m_sampleRate;
202 
203  SNDFILE* sndFile = sf_open(fileName.c_str(), SFM_WRITE, &sfInfo);
204 
205  if(!sndFile)
206  {
207  int result = sf_error(sndFile);
208 
209  switch (result)
210  {
211  case 0:
212  wxMessageBox("SF_ERR_NO_ERROR", fileName);
213  break;
214  case 1:
215  wxMessageBox("SF_ERR_UNRECOGNISED_FORMAT");
216  break;
217  case 2:
218  wxMessageBox("SF_ERR_SYSTEM", fileName);
219  break;
220  case 3:
221  wxMessageBox("SF_ERR_MALFORMED_FILE");
222  break;
223  case 4:
224  wxMessageBox("SF_ERR_UNSUPPORTED_ENCODING");
225  break;
226  default:
227  wxMessageBox("SOMETHING WENT AWRY");
228 
229  }
230 
231  return false;
232  }
233 
234  int delta = m_frameCount / c_waveDisplayWidth;
235 
236  int start = m_selStart * delta;
237  int frames = (m_selStop - m_selStart) * delta;
238 
239  sf_writef_double(sndFile, m_waveData.get() + (start * m_channels), frames);
240 
241  sf_write_sync(sndFile);
242 
243  sf_close(sndFile);
244 
245  return true;
246 }

250 void DetailDialog::LoadWavFile(wxString fileName)
251 {
252  SF_INFO sf_info;
253 
254  sf_info.format = 0;
255 
256  SNDFILE* snd_file = sf_open(fileName.c_str(), SFM_READ, &sf_info);
257 
258  if(snd_file)
259  {
262  m_frameCount = sf_info.frames;
263  m_channels = sf_info.channels;
264  m_sampleRate = sf_info.samplerate;
265  m_format = sf_info.format;
266 
267  if(m_channels < 1 || m_channels > 2)
268  {
269  wxMessageBox("Only mono and stereo channels are supported");
270  sf_close(snd_file);
271  return;
272  }
273 
274  m_waveData.reset(new double[m_frameCount * m_channels]);
275 
276  int read = sf_readf_double(snd_file, m_waveData.get(), m_frameCount);
277 
278  if(read != m_frameCount)
279  {
280  wxMessageBox("Read error");
281  return;
282  }
283 
288  wxMemoryDC dc;
289 
292  dc.SelectObject(*m_drawBuffer);
293 
298  wxString mapFile = fileName + ".bmp";
299 
300  if(wxFileExists(mapFile))
301  {
302  wxBitmap bitmap;
303  bitmap.LoadFile(mapFile, wxBITMAP_TYPE_BMP);
304  dc.DrawBitmap(bitmap, 0, 0);
305  }
306  else
307  {
310  SetCursor(wxCURSOR_WAIT);
311 
314  dc.SetBackground(*wxBLACK_BRUSH);
315 
318  dc.Clear();
319 
322  dc.SetPen(wxPen(*wxWHITE, 2, wxSOLID));
323  int prevX = 0;
324  int prevY = 50;
325 
326  double yDiv = (double) c_waveDisplayWidth / (double) m_frameCount;
327 
328  for(int idx = 0; idx < m_frameCount; ++idx)
329  {
330  double sample = (m_waveData.get())[idx * m_channels];
332  int newY = (int)(sample * 50.0) + 50;
333  int newX = (double)idx * yDiv;
334 
335  dc.DrawLine(prevX, prevY, newX, newY);
336 
337  prevX = newX;
338  prevY = newY;
339  }
340 
343  wxBitmap bitmap = dc.GetAsBitmap();
344  bitmap.SaveFile(mapFile, wxBITMAP_TYPE_BMP);
345 
348  SetCursor(wxCURSOR_DEFAULT);
349  }
350 
353  wxMemoryDC inverted_dc;
354 
355  inverted_dc.SelectObject(*m_invertedBuffer);
356 
357  inverted_dc.Blit
358  (
359  0,
360  0,
361  c_waveDisplayWidth,
362  c_waveDisplayHeight,
363  &dc,
364  0,
365  0,
366  wxSRC_INVERT,
367  false
368  );
369 
372  dc.SelectObject(wxNullBitmap);
373  inverted_dc.SelectObject(wxNullBitmap);
374 
377  m_selStart = c_minSel;
378  m_selStop = c_maxSel;
379 
382  Panel1->Refresh();
383  Panel1->Update();
384  }
385  else
386  {
387  int result = sf_error(snd_file);
388 
389  switch (result)
390  {
391  case 0:
392  wxMessageBox("SF_ERR_NO_ERROR", fileName);
393  break;
394  case 1:
395  wxMessageBox("SF_ERR_UNRECOGNISED_FORMAT");
396  break;
397  case 2:
398  wxMessageBox("SF_ERR_SYSTEM", fileName);
399  break;
400  case 3:
401  wxMessageBox("SF_ERR_MALFORMED_FILE");
402  break;
403  case 4:
404  wxMessageBox("SF_ERR_UNSUPPORTED_ENCODING");
405  break;
406  }
407  }
408 
409  sf_close(snd_file);
410 }

501 {
502  if(event.GetY() > (c_waveDisplayHeight * .5))

508  m_sound.Stop();

521  /*wxDragResult dragResult = */ dragSource.DoDragDrop();
522 
523  wxDELETE(data);
524  }
525  }
526  else
527  {
530  m_selecting = true;
531 
534  if(GetAsyncKeyState(VK_SHIFT))
535  {
536  m_sel2 = event.GetX();
537 
538  if(m_sel2 < m_selStart)
539  {
540  m_sel1 = m_selStop;
541  }
542  else if(m_sel2 > m_selStop)
543  {
544  m_sel1 = m_selStart;
545  }
546  }
547  else
548  {
549  m_sel1 = event.GetX();
550  }
551  }
552 }
